// Adicione estes models ao seu schema.prisma

model Account {
  id        String   @id @default(cuid())
  name      String
  type      String   @default("PERSON") // PERSON ou BUSINESS
  active    Boolean  @default(true)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

// Atualize o model PointsProgram para incluir o campo store:
model PointsProgram {
  id       String   @id @default(cuid())
  name     String   // Nome do programa (Esfera, Livelo, etc)
  store    String?  // Loja/Clube (Fast Shop, Casas Bahia, etc) - NOVO CAMPO
  type     String   @default("POINTS") // POINTS, MILES, CASHBACK
  active   Boolean  @default(true)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  accounts PointsAccount[]
}
